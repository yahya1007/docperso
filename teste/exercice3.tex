\begin{exercise} En utilisant la décomposition en fractions simples, calculer les sommes des séries suivantes:
	\choice
	{$\sum\limits_{n\geq 2}\frac{1}{n^{2}-1}$}
	{ $\sum\limits_{n\geq 1} \frac{1}{n(n+1)(n+2)}$}
	{\True $\sum\limits_{n\geq 0}\frac{9}{(3n+1)(3n+4)}$}
	{$\sum\limits_{n\geq 1} \frac{1}{n(n+1)(n+3)}$}
\end{exercise}
