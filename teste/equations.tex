\section{Un bloc d’équations avec \texttt{equation} et \texttt{aligned}}

\begin{equation*}
	\begin{aligned}
		      3x^2+4x+2 &= 2x^2+2x+6 \\
		3x^2-2x^2+4x-2x &= 6x-2 \\
		         x^2+2x &= 4 \\
		   x\times(x+2) &= 4
	\end{aligned}
\end{equation*}
