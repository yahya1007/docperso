\question[1] Soient $a$ et $b$ deux réels. Rappeler le développement de $\sin(a-b)$.
	\begin{solution}
		\begin{equation*}
			\sin\left(a-b\right)=\sin a \cos b-\cos a \sin b
		\end{equation*}
	\end{solution}
