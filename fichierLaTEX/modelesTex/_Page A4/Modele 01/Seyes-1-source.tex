\newcommand{\seyes}[2]{
  \ifthenelse{\equal{#1}{RL}}
    {
      \ifthenelse{\equal{#2}{1}}
         {\seyes@RL@I}{}
      \ifthenelse{\equal{#2}{2}}
         {\seyes@RL@II}{}
    }
   {\ifthenelse{\equal{#1}{LR}}
    {
      \ifthenelse{\equal{#2}{1}}
         {\seyes@LR@I}{}
      \ifthenelse{\equal{#2}{2}}
         {\seyes@LR@II}{}
    }}
}
