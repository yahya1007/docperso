\begin{exo}
\begin{questions}
\question $a$ et $b$ deux nombres rationnels tel que $a\leq 7$ et $3\leq b$.
\begin{parts}
\part Comparer $3a$ et $ab$ puis $7b$ et $ab$.
\part Déduire la comparaison entre $3a$ et $7b$.
\end{parts}
\question Soit $2\leq a \leq 6$ et $3\leq b \leq 9$.
\begin{parts}
\part Encadrer $2a+3$ ;; $3a+5b$ ;; $\dfrac{a}{2}+\dfrac{b}{3}$
\end{parts}
\question Soit $a$ et $b$ deux nombres rationnels tel que : $a\leq b$
\begin{parts}
\part Prouver que $2a\leq a+b \leq 2b$
\part Déduire que $a \leq \dfrac{a+b}{2} \leq b$
\end{parts}
\end{questions}
\end{exo}
